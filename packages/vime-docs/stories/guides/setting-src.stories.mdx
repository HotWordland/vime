# Setting Src

## Html5

media stream, dropbox url, src

```js
// Null
player.src = null;

// Relative
player.src = '/media/audio.mp3';

// Absolute
player.src= 'https://mysite.com/video.mp4';

// Object
player.src = { src: '/media/video.mp4', type: 'video/mp4' };

// Array of strings
player.src = [
  '/media/video.mp4',
  '/media/video.ogg'
];

// Array of objects
player.src = [{ 
  src: '/media/video.mp4', 
  type: 'video/mp4',
  quality:  
}, { 
  src: '/media/video.ogg', 
  type: 'video/ogg'
}];

// Passing in qualities
player.src = [{ 
  src: '/media/video-1080.mp4', 
  type: 'video/mp4' 
  quality: 1080
}, { 
  src: '/media/video-1080.ogg', 
  type: 'video/ogg',
  quality: 1080
}, { 
  src: '/media/video-720.mp4', 
  type: 'video/mp4' 
  quality: 720
}, { 
  src: '/media/video-720.ogg', 
  type: 'video/ogg',
  quality: 720
}];
```
